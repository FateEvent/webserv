#include <string>
#include <cmath>
#include <iostream>

static int	is_hex_digit(char c)
{
	if ((c >= 48 && c <= 57)
		|| (c >= 65 && c <= 70)
		|| (c >= 97 && c <= 102))
		return (1);
	return (0);
}

static size_t	getMaxPower(std::string hex)
{
	size_t	i;

	i = 0;
	while (hex[i])
	{
		if (is_hex_digit(hex[i]))
			i++;
		else
			break ;
	}
	--i;
	return (i);
}

ssize_t	hextol(std::string hex)
{
	size_t  i;
	size_t  power;
	ssize_t count;

	i = 0;
	count = 0;
	power = getMaxPower(hex);
	while (power >= 0)
	{
		if (hex[i] >= 97 && hex[i] <= 102)
			hex[i] -= 32;
		if (hex[i] >= 48 && hex[i] <= 57)
			count += (hex[i] - '0') * pow(16, power);
		else if (hex[i] >= 65 && hex[i] <= 70)
			count += (10 + hex[i] - 'A') * pow(16, power);
		else
			break ;
		power--;
		i++;
	}
	return (count);
}
